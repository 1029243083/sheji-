# 单例模式
- 保证一个类仅有一个实例，并提供一个访问它的全局访问点

```js
// 返回的始终是instent
const Test = (function () {
    let instent;
    return function (data) {
        if(instent) return instent;
        this.data = data;
        instent = this;
    }
})()
```


```js
const getSingFun = (function (func) {
    let sing;
    return function () {
        if(sing) return sing;
        sing = func.apply(this,arguments);
        return sing;
    }
})()
```
